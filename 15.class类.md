# 15.class类

## static

**静态成员只属于类本身，不属于实例化对象**

```js
function phone() {
	
}

phone.name = "手机"
phone.changge = function() {
    console.log("我可以唱歌");
}
phone.prototype.size = "5.5inch";
phone.prototype.call = function() {
    console.log("打call");
}

let nokia = new phone();
console.log(nokia.name);		// undefined
nokia.changge();

console.log(nokia.size);
nokia.call();
```

```js
class shouji {
    // 静态属性
    static name = "手机";
	// name = "手机";
	static changge() {
        console.log("我可以唱歌");
    }
}

// 类的实例化
let htcone = new shouji();
console.log(htcone.name);		// undefined
htcone.changge();				// 我可以唱歌
```
