# Symbol数据类型

**Symbol表示独一无二的值**

1. Symbol的值是唯一的，解决命名冲突的问题
```js
let s = Symbol();
console.log(s, typeof s);       // Symbol()  "symbol"
let s1 = Symbol("广科");
let s2 = Symbol("广科");
console.log(s1 === s2);         // false

let s3 = Symbol.for("广科");
console.log(s3, typeof s3);     // Symbol(广科)  "symbol"
let s4 = Symbol.for("广科");
console.log(s3 === s4);         // true
```


2. Symbol是不能进行运算的
```js
let result = s + 100;           // Cannot convert a symbol to a number
```

3. 不能使用<kbd>for...in...</kbd>来遍历循环


## test
```js
let obj = {
    up() {
        console.log("123");
    },
    down() {
        console.log("456");
    }
};

let methods = {
    up: Symbol(),
    down: Symbol()
};

obj[methods.up] = function() {
    console.log("我是up");
};
obj[methods.down] = function() {
    console.log("我是down");
};
console.log(obj);                   // {up: f,  down: f,  Symbol(): f,  Symbol(): f}
```

```js
let game = {
    name: "Apex",
    [Symbol("say")]: function() {
        console.log("我可以说话");
    },
    [Symbol("r301")]: function() {
        console.log("我需要R301");
    }
};
console.log(game);                  // {name: "Apex",  Symbol(say): f,  Symbol(r301): f}
```
