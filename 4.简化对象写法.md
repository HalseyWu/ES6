# 简化对象写法

```js
let school = "广科";
        let change = function () {
            console.log("这个世界发生了变化");
        }
        const xuexiao = {
            // 对象名: 对象值
            // school: school,
            // change: change,

            // 对象名: function () {...}
            // jianhua: function () {
            //     console.log("这个世界简化了")
            // },

            school,
            change,
            jianhua() {
                console.log("这个世界简化了")
            }
        }

        console.log(xuexiao.school)     // 广科
        xuexiao.jianhua()               // 这个世界简化了
```
