# 17.Get & Set

```js
class Phone {
    get price() {
        // 求和 求平均
        console.log("价格被获取");
        return "jiage";
    }
    set price(value) {
        // 做修改判断传过来的值是否符合要求
        console.log("价格被修改了");
        console.log(value);
    }
}

//实例化类
let s = new Phone();
// console.log(s.price);
s.price = "free";
```
